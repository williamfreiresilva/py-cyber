from pathlib import Path
from datetime import datetime, timedelta
import os

def find_recent_files(root: str, hours: int = 24, extensions: list = None):
    """Find files modified within the last N hours — useful for IR triage."""
    cutoff = datetime.now().timestamp() - (hours * 3600)
    results = []
    for path in Path(root).rglob("*"):
        if not path.is_file(): continue
        if extensions and path.suffix.lower() not in extensions: continue
        if path.stat().st_mtime >= cutoff:
            results.append({
                "path": str(path),
                "size_kb": round(path.stat().st_size / 1024, 2),
                "modified": datetime.fromtimestamp(path.stat().st_mtime).isoformat()
            })
    return results

# Usage: find .exe or .ps1 files modified in the last 2 hours
suspicious = find_recent_files("C:/Users", hours=2, extensions=[".exe", ".ps1", ".bat"])
for f in suspicious:
    print(f"[!] {f['modified']} | {f['size_kb']}KB | {f['path']}")

def preserve_evidence(source_path: str, case_id: str):
    """Copy artifact to timestamped evidence folder (never modify originals)."""
    ts = datetime.now().strftime("%Y%m%d_%H%M%S")
    dest_dir = Path(f"evidence/{case_id}/{ts}")

subprocess — Executing System Commands

Run system tools and capture output safely 
import subprocess

def run_command(cmd: list) -> str:
    """Execute a command and return stdout. Never use shell=True with user input."""
    result = subprocess.run(
        cmd,
        capture_output=True,
        text=True,
        timeout=30
    )
    if result.returncode != 0:
        print(f"[WARN] Command failed: {result.stderr}")
    return result.stdout

# Get running processes (Linux)
output = run_command(["ps", "aux"])

# Run netstat to check active connections (Windows)
connections = run_command(["netstat", "-ano"])

# Compute file hash using system tools
hash_out = run_command(["sha256sum", "/path/to/suspicious.exe"])
⚠️ Security Warning Never use shell=True with any input that comes from external sources. It enables command injection. Always pass commands as a list, not a string.

psutil — Process & Resource Monitoring

Detect anomalous processes during incident triage 
import psutil

def hunt_suspicious_processes():
    """List processes with high CPU or unusual names."""
    suspicious = []
    for proc in psutil.process_iter(['pid', 'name', 'cpu_percent', 'cmdline', 'username']):
        try:
            info = proc.info
            if info['cpu_percent'] > 80:
                suspicious.append(info)
            # Flag processes running from temp directories
            cmd = ' '.join(info['cmdline'] or []).lower()
            if any(x in cmd for x in ['temp', 'appdata', '\\tmp\\']):
                suspicious.append({**info, 'flag': 'temp_execution'})
        except (psutil.NoSuchProcess, psutil.AccessDenied):
            pass
    return suspicious

# System-wide resource snapshot
print(f"CPU: {psutil.cpu_percent()}% | RAM: {psutil.virtual_memory().percent}%")
print(f"Disk: {psutil.disk_usage('/').percent}% | Connections: {len(psutil.net_connections())}")

def export_excel_report(df: pd.DataFrame, filename: str):
    """Export DataFrame as a styled Excel workbook."""
    with pd.ExcelWriter(filename, engine="openpyxl") as writer:
        df.to_excel(writer, sheet_name="Triage Report", index=False)

            ws.column_dimensions[col[0].column_letter].width = 20
    print(f"[+] Report saved: {filename}")

from sqlalchemy import create_engine, text
import pandas as pd

# Connect — works with SQLite, PostgreSQL, MySQL, MSSQL

def get_asset_owner(ip_addresses: list) -> pd.DataFrame:
    """Look up asset owners for a list of IPs — use during alert triage."""
    placeholders = ",".join([f":ip{i}" for i in range(len(ip_addresses))])
    params = {f"ip{i}": ip for i, ip in enumerate(ip_addresses)}
    query = text(f"""
        SELECT ip, hostname, owner, department, criticality
        FROM assets
        WHERE ip IN ({placeholders})
    """)
    with engine.connect() as conn:
        return pd.read_sql(query, conn, params=params)

requests — Consuming REST APIs

Robust API client with retry logic and rate limiting 
import requests
import time
import os
from dotenv import load_dotenv

load_dotenv()

class VirusTotalClient:
    BASE_URL = "https://www.virustotal.com/api/v3"

    def __init__(self):
        self.session = requests.Session()
        self.session.headers.update({
            "x-apikey": os.getenv("VT_API_KEY"),
            "Accept": "application/json"
        })

    def _get(self, endpoint: str, retries=3) -> dict:
        for attempt in range(retries):
            try:
                r = self.session.get(f"{self.BASE_URL}{endpoint}", timeout=15)
                if r.status_code == 429:          # rate limited
                    time.sleep(60); continue
                r.raise_for_status()
                return r.json()
            except requests.RequestException as e:
                if attempt == retries - 1: raise
                time.sleep(2 ** attempt)     # exponential back-off

    def lookup_hash(self, file_hash: str) -> dict:
        data = self._get(f"/files/{file_hash}")
        stats = data["data"]["attributes"]["last_analysis_stats"]
        return {
            "hash": file_hash,
            "malicious": stats["malicious"],
            "suspicious": stats["suspicious"],
            "verdict": "MALICIOUS" if stats["malicious"] > 3 else "CLEAN"
        }

    def lookup_ip(self, ip: str) -> dict:
        data = self._get(f"/ip_addresses/{ip}")
        attr = data["data"]["attributes"]
        return {"ip": ip, "country": attr.get("country"), "asn": attr.get("asn")}

from pydantic import BaseModel, field_validator, ValidationError
from typing import Optional, Literal
from datetime import datetime

class IOCResult(BaseModel):
    ioc: str
    ioc_type: Literal["hash", "ip", "domain", "url"]
    malicious_count: int
    verdict: Literal["CLEAN", "SUSPICIOUS", "MALICIOUS"]
    source: str
    queried_at: datetime = datetime.now()
    tags: Optional[list[str]] = []

    @field_validator("malicious_count")
    def must_be_positive(cls, v):
        if v < 0:
            raise ValueError("malicious_count cannot be negative")
        return v

# Parse and validate API response
try:

        ioc_type="hash",
        malicious_count=14,
        verdict="MALICIOUS",
        source="VirusTotal"
    )
    print(result.model_dump_json(indent=2))
except ValidationError as e:

Real Examples: VirusTotal, CrowdStrike, QRadar

CrowdStrike Falcon — query detections via API 
import requests, os

class CrowdStrikeClient:
    BASE = "https://api.crowdstrike.com"

    def __init__(self):
        self.token = self._authenticate()

    def _authenticate(self) -> str:
        r = requests.post(f"{self.BASE}/oauth2/token", data={
            "client_id": os.getenv("CS_CLIENT_ID"),
            "client_secret": os.getenv("CS_CLIENT_SECRET")
        })
        return r.json()["access_token"]

    def get_detections(self, severity="High") -> list:
        headers = {"Authorization": f"Bearer {self.token}"}
        # Step 1: Get detection IDs
        r = requests.get(f"{self.BASE}/detects/queries/detects/v1",
            headers=headers,
            params={"filter": f"max_severity_displayname:'{severity}'", "limit": 100})
        ids = r.json().get("resources", [])
        if not ids: return []
        # Step 2: Get detection details
        r2 = requests.post(f"{self.BASE}/detects/entities/summaries/GET/v1",
            headers=headers, json={"ids": ids})
        return r2.json().get("resources", [])

import json
import subprocess

def run_bandit_scan(target_dir: str) -> list:
    """Run bandit and return HIGH/MEDIUM findings."""
    result = subprocess.run(
        ["bandit", "-r", target_dir, "-f", "json", "-ll"],
        capture_output=True, text=True
    )
    try:
        report = json.loads(result.stdout)
    except json.JSONDecodeError:
        return []

    findings = []
    for issue in report.get("results", []):
        if issue["issue_severity"] in ["HIGH", "MEDIUM"]:
            findings.append({
                "file": issue["filename"],
                "line": issue["line_number"],
                "severity": issue["issue_severity"],
                "issue": issue["issue_text"],
                "test_id": issue["test_id"]
            })
    return findings

issues = run_bandit_scan("./src")
for i in issues:
    print(f"[{i['severity']}] {i['file']}:{i['line']} — {i['issue']}")

from bs4 import BeautifulSoup
import re

def scrape_threat_page(url: str) -> dict:
    """Extract IOCs (IPs, hashes, domains) from a threat report page."""
    headers = {"User-Agent": "Mozilla/5.0 (Security Research)"}
    r = requests.get(url, headers=headers, timeout=15)

    text = soup.get_text()
    return {
        "ips": re.findall(r'\b(?:\d{1,3}\.){3}\d{1,3}\b', text),
        "md5s": re.findall(r'\b[a-fA-F0-9]{32}\b', text),

        "domains": re.findall(r'\b(?:[a-z0-9\-]+\.)+(?:com|net|org|io|xyz|ru)\b', text)
    }

OWASP ZAP API — Automated Vulnerability Scanning

Trigger and retrieve ZAP scan results programmatically 
import requests
import time

class ZAPScanner:
    def __init__(self, zap_url="http://localhost:8080", api_key=""):
        self.base = f"{zap_url}/JSON"
        self.key = api_key

    def _req(self, component, action, params={}) -> dict:
        params["apikey"] = self.key
        return requests.get(f"{self.base}/{component}/action/{action}",
                            params=params).json()

    def spider_and_scan(self, target_url: str) -> list:
        """Run spider then active scan, return HIGH findings."""
        # Start spider
        spider_id = self._req("spider", "scan", {"url": target_url})["scan"]
        while True:
            progress = int(requests.get(f"{self.base}/spider/view/status",
                params={"scanId": spider_id, "apikey": self.key}).json()["status"])
            if progress >= 100: break
            time.sleep(3)
        # Active scan
        scan_id = self._req("ascan", "scan", {"url": target_url})["scan"]
        while True:
            status = int(requests.get(f"{self.base}/ascan/view/status",
                params={"scanId": scan_id, "apikey": self.key}).json()["status"])
            if status >= 100: break
            time.sleep(5)
        # Get alerts
        alerts = requests.get(f"{self.base}/alert/view/alerts",
            params={"baseurl": target_url, "apikey": self.key}).json()["alerts"]
        return [a for a in alerts if a["risk"] in ["High", "Medium"]]

Hash files for IOC matching and evidence integrity 
import hashlib
from pathlib import Path
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.backends import default_backend

def hash_file(filepath: str) -> dict:
    """Compute MD5, SHA1, SHA256 of a file — standard for IR evidence."""
    md5 = hashlib.md5()
    sha1 = hashlib.sha1()
    sha256 = hashlib.sha256()

    with open(filepath, "rb") as f:
        for chunk in iter(lambda: f.read(65536), b""):
            md5.update(chunk)
            sha1.update(chunk)
            sha256.update(chunk)

    return {
        "file": str(filepath),
        "md5": md5.hexdigest(),
        "sha1": sha1.hexdigest(),
        "sha256": sha256.hexdigest(),
        "size_bytes": Path(filepath)        "size_bytes": Path(filepath).stat().st_size
    }

def hash_directory(dirpath: str, extensions=None) -> list:
    """Hash all files in a directory — for bulk IOC matching."""
    results = []
    for p in Path(dirpath).rglob("*"):
        if not p.is_file(): continue
        if extensions and p.suffix.lower() not in extensions: continue
        try:
            results.append(hash_file(str(p)))
        except (PermissionError, OSError):
            pass
    return results

datetime — Forensic Timelines & Timezone Handling

⚠️ Forensic Rule Always store and compare timestamps in UTC. Convert to local time only for display. Inconsistent timezone handling is one of the most common causes of incorrect forensic timelines.
Build a normalized forensic timeline from multiple log sources 
from datetime import datetime, timezone
from dateutil import parser as dateparser
import pytz

def normalize_timestamp(ts_string: str, source_tz: str = "UTC") -> str:
    """Parse any timestamp format and return ISO 8601 UTC string."""
    try:
        # dateutil handles almost any format automatically
        dt = dateparser.parse(ts_string)
        if dt.tzinfo is None:
            # Localize naive timestamps to their source timezone
            tz = pytz.timezone(source_tz)
            dt = tz.localize(dt)
        # Convert everything to UTC
        return dt.astimezone(timezone.utc).isoformat()
    except Exception as e:
        return f"PARSE_ERROR: {e}"

def build_timeline(events: list) -> list:
    """Sort multi-source events into a unified UTC timeline."""
    for event in events:
        event["utc_time"] = normalize_timestamp(
            event["raw_time"],
            event.get("source_tz", "UTC")
        )
    return sorted(events, key=lambda x: x["utc_time"])

# Example: merge Windows Event Log + Linux syslog + firewall log
events = [
    {"source": "WinEvt",   "raw_time": "2025-06-10 14:32:01", "source_tz": "America/Sao_Paulo", "event": "Logon 4624"},
    {"source": "Syslog",    "raw_time": "Jun 10 17:32:05",    "source_tz": "UTC",                  "event": "sudo su"},
    {"source": "Firewall",  "raw_time": "1749573128",         "source_tz": "UTC",                  "event": "DENY outbound 4444"},
]
timeline = build_timeline(events)
for e in timeline:

from scapy.all import rdpcap, IP, TCP, UDP, DNS, DNSQR
from collections import Counter

def analyze_pcap(pcap_file: str) -> dict:
    """Extract threat indicators from a packet capture file."""
    packets = rdpcap(pcap_file)
    dns_queries, connections, large_transfers = [], [], []

    for pkt in packets:
        if pkt.haslayer(DNS) and pkt.haslayer(DNSQR):
            qname = pkt[DNSQR].qname.decode("utf-8", errors="ignore").rstrip(".")
            dns_queries.append(qname)

        if pkt.haslayer(IP) and pkt.haslayer(TCP):
            conn = (pkt[IP].src, pkt[IP].dst, pkt[TCP].dport)
            connections.append(conn)
            if len(pkt) > 8192:
                large_transfers.append(conn)

    return {
        "top_dns_queries": Counter(dns_queries).most_common(20),
        "top_destinations": Counter([c[1] for c in connections]).most_common(10),
        "suspicious_ports": [c for c in connections if c[2] in {4444,1337,8888,9001,6667}],
        "large_transfers": large_transfers[:10],
        "total_packets": len(packets)
    }

paramiko — SSH Automation for Remote Response

Execute remote IR commands via SSH 
import paramiko
import os

class RemoteResponder:
    """Execute IR commands on remote Linux hosts via SSH."""

    def __init__(self, host: str, username: str, key_path: str):
        self.client = paramiko.SSHClient()
        self.client.set_missing_host_key_policy(paramiko.RejectPolicy())
        self.client.connect(host, username=username,
            key_filename=key_path, timeout=10)

    def run(self, command: str) -> str:
        stdin, stdout, stderr = self.client.exec_command(command, timeout=30)
        return stdout.read().decode().strip()

    def collect_artifacts(self) -> dict:
        """Collect key IR data from a live Linux host."""
        return {
            "processes":    self.run("ps aux --sort=-%cpu | head -20"),
            "connections":  self.run("ss -tulnp"),
            "crontabs":     self.run("crontab -l 2>/dev/null; ls /etc/cron*"),
            "last_logins":  self.run("last -20"),
            "bash_history": self.run("cat ~/.bash_history | tail -50"),
            "suid_files":   self.run("find / -perm -4000 -type f 2>/dev/null"),
        }

    def isolate_host(self, confirm: bool = False):
        """Block all traffic except SSH — use with caution during containment."""
        if not confirm:
            raise ValueError("Set confirm=True to execute host isolation")
        self.run("iptables -P INPUT DROP; iptables -P FORWARD DROP; iptables -P OUTPUT DROP")
        self.run("iptables -A INPUT -p tcp --dport 22 -j ACCEPT")
        self.run("iptables -A OUTPUT -p tcp --sport 22 -j ACCEPT")

    def __del__(self):
        self.client.close()
🚨 Operational Warning Always use RejectPolicy() — never AutoAddPolicy(). Add SSH host keys to ~/.ssh/known_hosts before connecting. Using AutoAddPolicy exposes you to man-in-the-middle attacks during an active incident.

import asyncio

# ── Core concept: coroutine ──
async def fetch_ioc(ioc: str) -> dict:
    await asyncio.sleep(0.1)   # simulates I/O wait — never blocks event loop
    return {"ioc": ioc, "verdict": "clean"}

# ── Run multiple coroutines concurrently ──
async def enrich_batch(iocs: list) -> list:
    tasks = [fetch_ioc(ioc) for ioc in iocs]
    return await asyncio.gather(*tasks, return_exceptions=True)

# ── Semaphore: rate limiting — never hammer APIs ──
async def enrich_with_limit(iocs: list, max_concurrent=20) -> list:
    sem = asyncio.Semaphore(max_concurrent)

    async def limited_fetch(ioc):
        async with sem:
            return await fetch_ioc(ioc)

    return await asyncio.gather(*[limited_fetch(i) for i in iocs])

# ── Entry point ──
if __name__ == "__main__":
    ioc_list = [f"1.2.3.{i}" for i in range(100)]
    results = asyncio.run(enrich_with_limit(ioc_list))

aiohttp — Async HTTP for Large-Scale Telemetry

Production-grade async HTTP client with timeout and retry 
import aiohttp
import asyncio
import os
from typing import Optional

async def async_vt_lookup(
    session: aiohttp.ClientSession,
    ioc: str,
    ioc_type: str = "files"
) -> dict:
    """Single async IOC lookup against VirusTotal."""
    url = f"https://www.virustotal.com/api/v3/{ioc_type}/{ioc}"
    try:
        async with session.get(url, timeout=aiohttp.ClientTimeout(total=15)) as resp:
            if resp.status == 200:
                data = await resp.json()
                stats = data["data"]["attributes"]["last_analysis_stats"]
                return {"ioc": ioc, "malicious": stats["malicious"], "status": "ok"}
            elif resp.status == 404:
                return {"ioc": ioc, "malicious": 0, "status": "not_found"}
            else:
                return {"ioc": ioc, "status": f"http_{resp.status}"}
    except (aiohttp.ClientError, asyncio.TimeoutError) as e:
        return {"ioc": ioc, "status": "error", "detail": str(e)}

async def bulk_vt_enrich(iocs: list, max_concurrent: int = 10) -> list:
    """Enrich a list of IOCs against VT with rate limiting."""
    headers = {"x-apikey": os.getenv("VT_API_KEY")}
    sem = asyncio.Semaphore(max_concurrent)

    async with aiohttp.ClientSession(headers=headers) as session:
        async def limited(ioc):
            async with sem:
                result = await async_vt_lookup(session, ioc)
                await asyncio.sleep(0.15)  # VT free tier: ~4 req/sec
                return result

        return await asyncio.gather(*[limited(ioc) for ioc in iocs])

            async with sem:
                return await enrich_ip_full(session, ip)
        return await asyncio.gather(*[limited(ip) for ip in ip_list])

import asyncio, aiohttp, json, os
import pandas as pd
from pydantic import BaseModel, ValidationError
from datetime import datetime
from pathlib import Path
from dotenv import load_dotenv

load_dotenv()

class EnrichmentResult(BaseModel):
    ioc: str
    malicious: int = 0
    suspicious: int = 0
    harmless: int = 0
    verdict: str = "UNKNOWN"
    source: str = "VirusTotal"
    enriched_at: str = ""

async def enrich_one(session: aiohttp.ClientSession, sem: asyncio.Semaphore, ioc: str) -> dict:
    async with sem:
        try:
            url = f"https://www.virustotal.com/api/v3/ip_addresses/{ioc}"
            async with session.get(url, timeout=aiohttp.ClientTimeout(total=15)) as r:
                if r.status != 200:
                    return {"ioc": ioc, "verdict": "NO_DATA"}
                data = await r.json()
                stats = data["data"]["attributes"]["last_analysis_stats"]
                mal = stats["malicious"]
                return EnrichmentResult(
                    ioc=ioc,
                    malicious=mal,
                    suspicious=stats["suspicious"],
                    harmless=stats["harmless"],
                    verdict="MALICIOUS" if mal > 3 else "SUSPICIOUS" if mal > 0 else "CLEAN",
                    enriched_at=datetime.utcnow().isoformat()
                ).model_dump()
        except Exception as e:
            return {"ioc": ioc, "verdict": "ERROR", "detail": str(e)}
        finally:
            await asyncio.sleep(0.15)

async def run_pipeline(ioc_file: str):
    iocs = list({l.strip() for l in Path(ioc_file).read_text().splitlines() if l.strip()})
    print(f"[*] Enriching {len(iocs)} unique IOCs...")
    sem = asyncio.Semaphore(10)
    headers = {"x-apikey": os.getenv("VT_API_KEY")}
    async with aiohttp.ClientSession(headers=headers) as session:
        results = await asyncio.gather(*[enrich_one(session, sem, ioc) for ioc in iocs])

    df = pd.DataFrame(results).sort_values("malicious", ascending=False)
    df.to_json("output/findings.json", orient="records", indent=2)
    df.to_excel("output/ioc_report.xlsx", index=False)
    print(f"[+] Done. Malicious: {len(df[df.verdict=='MALICIOUS'])} / {len(df)}")

if __name__ == "__main__":
    asyncio.run(run_pipeline("data/iocs.txt"))

import re, json
import pandas as pd
from pathlib import Path
from collections import defaultdict

# ── Scoring rules — customise per your environment ──
SCORE_RULES = {
    r"failed password":           (10, "Brute Force"),
    r"accepted publickey":        (5,  "SSH Login"),
    r"sudo:\s+\w+.*COMMAND":       (15, "Privilege Escalation"),
    r"useradd|groupadd":           (20, "Account Created"),
    r"iptables|ufw":               (10, "Firewall Change"),
    r"/etc/passwd|/etc/shadow":    (25, "Credential Access"),
    r"base64|curl.*\|\s*bash":    (30, "C2 Execution Pattern"),
}

def score_log_line(line: str) -> tuple:
    total_score, matched_rules = 0, []
    for pattern, (score, label) in SCORE_RULES.items():
        if re.search(pattern, line, re.I):
            total_score += score
            matched_rules.append(label)
    return total_score, matched_rules

def classify_severity(score: int) -> str:
    if score >= 30: return "CRITICAL"
    if score >= 20: return "HIGH"
    if score >= 10: return "MEDIUM"
    return "LOW"

IP_RE = re.compile(r'\b(?:\d{1,3}\.){3}\d{1,3}\b')

def triage_log_file(logpath: str) -> pd.DataFrame:
    records = []
    with open(logpath, "r", errors="replace") as f:
        for lineno, line in enumerate(f, 1):
            score, rules = score_log_line(line)
            if score == 0: continue
            ips = IP_RE.findall(line)
            records.append({
                "line": lineno, "score": score,
                "severity": classify_severity(score),
                "rules": ", ".join(rules),
                "src_ip": ips[0] if ips else "",
                "raw": line.strip()[:120]
            })
    return pd.DataFrame(records).sort_values("score", ascending=False)

import hashlib, shutil, json
from pathlib import Path
from datetime import datetime, timezone

# Evidence targets — adapt per OS and incident type
EVIDENCE_TARGETS = {
    "linux": [
        "/var/log/auth.log", "/var/log/syslog",
        "/etc/passwd", "/etc/crontab",
        "/root/.bash_history", "/tmp",
    ],
    "windows": [
        "C:/Windows/System32/winevt/Logs",
        "C:/Windows/Prefetch",
        "C:/Users",
    ]
}

def sha256_file(path: Path) -> str:
    h = hashlib.sha256()
    with open(path, "rb") as f:
        for chunk in iter(lambda: f.read(65536), b""):
            h.update(chunk)
    return h.hexdigest()

def collect_evidence(case_id: str, platform: str = "linux"):
    ts = datetime.now(timezone.utc).strftime("%Y%m%dT%H%M%SZ")
    out_dir = Path(f"evidence/{case_id}/{ts}")
    out_dir.mkdir(parents=True, exist_ok=True)
    manifest = []

    for target in EVIDENCE_TARGETS.get(platform, []):
        src = Path(target)        if not src.exists():
            manifest.append({"source": target, "status": "NOT_FOUND"})
            continue
        try:
            if src.is_file():
                dest = out_dir / src.name
                shutil.copy2(src, dest)
                manifest.append({
                    "source": target,
                    "dest": str(dest),
                    "sha256": sha256_file(dest),
                    "size_bytes": dest.stat().st_size,
                    "status": "COLLECTED"
                })
            elif src.is_dir():
                dest_dir = out_dir / src.name
                shutil.copytree(src, dest_dir, dirs_exist_ok=True)
                manifest.append({"source": target, "dest": str(dest_dir), "status": "DIR_COLLECTED"})
        except (PermissionError, OSError) as e:
            manifest.append({"source": target, "status": "ERROR", "detail": str(e)})

    # Write chain-of-custody manifest
    manifest_path = out_dir / "manifest.json"
    manifest_path.write_text(json.dumps({
        "case_id": case_id,
        "collected_at_utc": ts,
        "platform": platform,
        "artifacts": manifest
    }, indent=2))

    # Package everything into a zip archive
    archive = shutil.make_archive(f"evidence/{case_id}_{ts}", "zip", str(out_dir))
    print(f"[+] Evidence bundle: {archive}")
    print(f"[+] Artifacts collected: {len([m for m in manifest if 'COLLECTED' in m['status']])}")
    return manifest

if __name__ == "__main__":
    collect_evidence(case_id="IR-2025-042", platform="linux")

  • Document your scripts — a well-documented tool is worth ten undocumented ones. Write a README with installation, usage, and examples.
  • Automate your current job first — identify a repetitive task in your current role and automate it. Concrete impact beats theoretical knowledge.
  • Contribute to open-source security tools — even documentation fixes on projects like Volatility, MISP, or Sigma are valuable.
  • Write about what you build — blog posts, LinkedIn articles, or internal wiki pages. Teaching consolidates your own learning.
  • Learn to read, not just write — study the source code of tools you use daily. Understanding how scapy or requests work internally makes you a better user.
🐍 Final Word The goal is not to become a software engineer — it is to become an analyst who can make computers do the repetitive work, so you can focus on the thinking that only humans can do. Start small, automate one thing at a time, and compound your skills every week.
