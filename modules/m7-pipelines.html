CAPSTONE

O que é um pipeline de SOC?

Pipeline é a cola que liga as peças que você aprendeu nos módulos anteriores em um fluxo de produção: entra log/alerta bruto, sai decisão acionável (bloqueio, ticket, e-mail). 

FLOW┌──────────┐   ┌──────────┐   ┌──────────┐   ┌──────────┐   ┌──────────┐
│ INGESTÃO │ → │  PARSE   │ → │ ENRICH   │ → │  TRIAGE  │ → │  OUTPUT  │
│ log/CSV  │   │ regex/df │   │ VT/geo   │   │ score/IA │   │ XLSX/API │
└──────────┘   └──────────┘   └──────────┘   └──────────┘   └──────────┘
   M1/M2          M2/M5          M3/M6           M2            M2/M3
PRINCÍPIOS
  • Cada etapa é função pura testável
  • Falha em uma fonte ≠ pipeline parado
  • Logs estruturados (JSON, não print)
  • Idempotente — rerun não duplica
  • Config separada do código

Pipeline 1 — IOC Enrichment

PYTHONimport asyncio, json
from pyodide.http import pyfetch
from pydantic import BaseModel
import pandas as pd

class IOC(BaseModel):
    ip: str
    country: str = ""
    org: str = ""
    risk: int = 0

async def enrich_one(ip: str, sem):
    async with sem:
        try:
            r = await asyncio.wait_for(
                pyfetch(f"https://ipapi.co/{ip}/json/"),
                timeout=5.0
            )
            j = await r.json()
            risk = 10 if "Tor" in j.get("org", "") else 0
            return IOC(ip=ip, country=j.get("country_name", ""),
                       org=j.get("org", ""), risk=risk)
        except Exception:
            return IOC(ip=ip)

async def enrich_batch(ips, concurrency=5):
    sem = asyncio.Semaphore(concurrency)
    results = await asyncio.gather(*[enrich_one(i, sem) for i in ips])
    return pd.DataFrame([r.model_dump() for r in results])

ips = ["8.8.8.8", "1.1.1.1", "45.33.32.156"]
df = await enrich_batch(ips)
df

Pipeline 2 — Alert Triage com scoring

PYTHONimport pandas as pd
import re
from datetime import datetime, timezone

# Alertas crus do SIEM
alerts = [
    {"id": "A1", "src_ip": "45.33.32.156", "sig": "SSH brute force", "count": 120},
    {"id": "A2", "src_ip": "10.0.0.5",     "sig": "Failed login",    "count": 3},
    {"id": "A3", "src_ip": "185.220.101.5","sig": "C2 beacon",       "count": 5},
    {"id": "A4", "src_ip": "192.168.1.7",  "sig": "Port scan",       "count": 800},
]

def score(a: dict) -> int:
    s = 0
    if "C2" in a["sig"]: s += 50
    if "brute" in a["sig"].lower(): s += 30
    if a["count"] > 100: s += 20
    if a["src_ip"].startswith(("10.", "192.168.")): s -= 10
    return max(0, s)

def priority(s: int) -> str:
    if s >= 50: return "P1"
    if s >= 25: return "P2"
    return "P3"

df = pd.DataFrame(alerts)
df["score"] = df.apply(score, axis=1)
df["priority"] = df["score"].apply(priority)
df = df.sort_values("score", ascending=False)
df

Pipeline 3 — Forensic Collector

PYTHONimport hashlib, json
from datetime import datetime, timezone
from cryptography.fernet import Fernet

def collect_evidence(label: str, content: bytes, key: bytes):
    # 1. cadeia de custódia
    custody = {
        "label": label,
        "size": len(content),
        "sha256": hashlib.sha256(content).hexdigest(),
        "collected_at": datetime.now(timezone.utc).isoformat(),
    }

    # 2. cifrar conteúdo
    cipher = Fernet(key)
    encrypted = cipher.encrypt(content)

    # 3. assinar metadados
    custody["encrypted_size"] = len(encrypted)
    custody["signature"] = hashlib.sha256(encrypted).hexdigest()

    return {"custody": custody, "payload": encrypted}

key = Fernet.generate_key()
ev = collect_evidence("memdump_srv01", b"", key)
print(json.dumps(ev["custody"], indent=2))

Logging estruturado

PYTHONimport logging, json, sys
from datetime import datetime, timezone

class JSONFormatter(logging.Formatter):
    def format(self, record):
        return json.dumps({
            "ts": datetime.now(timezone.utc).isoformat(),
            "level": record.levelname,
            "msg": record.getMessage(),
            "module": record.module,
        })

logger = logging.getLogger("pipeline")
logger.setLevel(logging.INFO)
h = logging.StreamHandler(sys.stdout)
h.setFormatter(JSONFormatter())
logger.addHandler(h)

logger.info("pipeline started")
logger.warning("rate limit hit, retrying")

DICA DE CAMPO Logs estruturados em JSON são parseáveis pelo SIEM da sua própria empresa — é meta, mas é assim que pipelines maduros funcionam. Splunk, Elastic e Datadog tratam JSON nativamente.

Capstone — pipeline completo end-to-end

Combine os 3 pipelines em um só fluxo: receba lista de alertas (com src_ip), enriqueça os IPs com geo, calcule score, gere XLSX com 2 abas (all_alerts, p1_only), e logue cada etapa em JSON. Esse é o entregável final — o que diferencia analista júnior de pleno. 

PYTHONasync def soc_pipeline(alerts: list, key: bytes):
    logger.info(f"started: {len(alerts)} alerts")

    # 1. enrich
    ips = [a["src_ip"] for a in alerts]
    enriched = await enrich_batch(ips, concurrency=5)
    logger.info("enrichment done")

    # 2. score + priority
    df = pd.DataFrame(alerts)
    df = df.merge(enriched, left_on="src_ip", right_on="ip", how="left")
    df["score"] = df.apply(score, axis=1)
    df["priority"] = df["score"].apply(priority)

    # 3. output
    with pd.ExcelWriter("/tmp/soc_report.xlsx") as w:
        df.to_excel(w, sheet_name="all_alerts", index=False)
        df[df["priority"] == "P1"].to_excel(w, sheet_name="p1_only", index=False)
    logger.info("report saved")
    return df

result = await soc_pipeline(alerts, key)
result
